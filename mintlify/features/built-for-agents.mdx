### Developer Integration Patterns

<Tabs>
  <Tab title="Node.js Integration">
    ```typescript
    import { FastMCP } from 'fastmcp';
    
    const mcp = new FastMCP({
      server: 'https://api.agentic.scope3.com/mcp',
      headers: {
        'x-scope3-api-key': process.env.SCOPE3_API_KEY
      }
    });
    
    // Your agent can now call Scope3 tools
    const campaign = await mcp.call('create_campaign', {
      brandAgentId: 'ba_123',
      name: 'AI-Generated Campaign',
      prompt: 'Target high-intent users with video creative',
      budget: { total: 10000, currency: 'USD' }
    });
    ```
  </Tab>
  
  <Tab title="Python Integration">
    ```python
    import mcp_client
    
    client = mcp_client.Client(
        server_url='https://api.agentic.scope3.com/mcp',
        headers={'x-scope3-api-key': os.environ['SCOPE3_API_KEY']}
    )
    
    # Create campaigns programmatically
    campaign = client.call_tool('create_campaign', {
        'brandAgentId': 'ba_123',
        'name': 'AI-Optimized Campaign',
        'prompt': 'Focus on premium inventory with strong completion rates',
        'budget': {'total': 15000, 'currency': 'USD'}
    })
    ```
  </Tab>
  
  <Tab title="Agent Framework">
    ```javascript
    // Example agent using our MCP endpoints
    class AdvertisingAgent {
      constructor(scope3MCP) {
        this.scope3 = scope3MCP;
      }
      
      async optimizeCampaigns(brandAgentId) {
        // Get campaign performance
        const campaigns = await this.scope3.call('list_campaigns', { brandAgentId });
        
        // Analyze tactics for optimization opportunities  
        for (const campaign of campaigns) {
          const analysis = await this.scope3.call('analyze_tactics', {
            campaignId: campaign.id,
            analysisType: 'efficiency'
          });
          
          // Make optimization decisions based on AI analysis
          if (analysis.recommendations.length > 0) {
            await this.implementOptimizations(campaign, analysis);
          }
        }
      }
    }
    ```
  </Tab>
</Tabs>
